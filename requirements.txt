# Python dependencies for the Capkala PETI investigation pipeline.
# Install: pip install -r requirements.txt

# Core geospatial processing
rasterio               # Read/write GeoTIFFs
numpy                  # Array math
Pillow                 # PNG output
requests               # HTTP downloads

# Backend A — Google Earth Engine (detect.py --backend gee, the default)
earthengine-api        # GEE Python API

# Backend B — Microsoft Planetary Computer (detect.py --backend mpc, no account)
pystac-client          # Query the Planetary Computer STAC API
planetary-computer     # Sign asset URLs (anonymous)
odc-stac               # Load STAC items into xarray
rioxarray              # xarray <-> raster helpers

# Map products (detect.py --map, make_map.py)
matplotlib             # Map layout / rendering
contextily             # OpenStreetMap basemap tiles

# Text-to-speech (video narration)
elevenlabs             # ElevenLabs TTS (Bian voice)

# Note: video assembly (scripts/02_assemble_video.py) additionally requires
# ffmpeg + ffprobe installed as system binaries (e.g. `brew install ffmpeg`).
